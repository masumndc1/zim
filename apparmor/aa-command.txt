#include <tunables/global>

/usr/sbin/apache2 {
  #include <abstractions/apache2-common>
  #include <abstractions/base>
  #include <abstractions/lxc/container-base>
  #include <abstractions/nis>

  deny /var/www/html/deny/* r,

  ^DEFAULT_URI {
  }

  ^HANDLING_UNTRUSTED_INPUT {
  }
}
