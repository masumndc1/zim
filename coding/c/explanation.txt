

-- example of a variable declaration;

	void (*func)(const Arg *arg);

void (*func)(const Arg *arg); declares a variable named func that is a pointer to a function.
That function takes one argument: const Arg *
and returns void


`(*func)`
The parentheses matter.
func is not a function — it is a pointer to a function.
Without parentheses, void *func(const Arg *arg); would mean something completely
different (a function returning a void*).

-- another one:

static int (*xerrorxlib)(Display *, XErrorEvent *);

declares a static variable named xerrorxlib that is a pointer to a function.
That function takes:
a Display * , and a XErrorEvent *, and returns an int.

-- how to create custom return type:

#include <stdio.h>
#include <stdlib.h>

typedef struct Monitor {
    int id;
} Monitor;

Monitor *createmon(void) {
    Monitor *m = malloc(sizeof(Monitor));
    m->id = 1;
    return m;
}

int main(void) {
    Monitor *m = createmon();
    printf("Monitor id = %d\n", m->id);
    free(m);
}
