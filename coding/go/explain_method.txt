/*
methods
it is not a requirement to write capital letters for functions in Go—you
only capitalize a function name if you want it to be public (exported) so
other files outside its current directory can use it. Lowercase functions
are completely valid but remain private (unexported), meaning they can only
be used by files inside that exact same package folder
*/

package main

import "fmt"

// 1. Define a simple struct data type
type User struct {
	Name string
}

// 2. Attach a method to the struct using the receiver '(u User)'
func (u User) Greet() {
	// 'u' lets us access the data inside the struct instance
	fmt.Printf("Hello, my name is %s!\n", u.Name)
}

/* output
hello my name is alice
*/
package main

import (
	"fmt"
)

type Person struct {
	Name string
}

func (p Person) greet() {
	fmt.Printf("hello my name is %s\n", p.Name)
}
