... in go, functions are first class citizen. ie, functions
... can be treated as regular variables, bool etc.

func fakeSensor() kelvin {
	return kelvin(rand.Intn(151) + 150)
}
func realSensor() kelvin {
	return 0
}

... now assign function to a variable and omit () of function
... when assining.

func main() {
	sensor := fakeSensor
	fmt.Println(sensor())

	sensor = realSensor
	fmt.Println(sensor())
}

... example of passing function as parameter to another function
... generated by ai

package main

import "fmt"

// 1. This is a helper function

func checkFirewall() bool {
	fmt.Println("Checking iptables firewall rules...")
	return true
}

// 2. This function accepts another function as its second parameter
// Look at the type definition: 'task func() bool'

func ExecuteAudit(nodeName string, task func() bool) {
	fmt.Printf("--- Starting Security Audit on: %s ---\n", nodeName)

	// Execute the function that
	// was passed in as a parameter

	var isSecure bool
	isSecure = task()

	if isSecure {
		fmt.Println("Status: Node is compliant.")
	} else {
		fmt.Println("Status: ALERT! Fix required.")
	}
}

func main() {
	// 3. Pass the 'checkFirewall' function directly as a parameter into 'ExecuteAudit'
	// Notice we do NOT include parentheses '()' when passing it;
	// we are passing the function itself!

	ExecuteAudit("compute-node-01", checkFirewall)
}
