package main

import "fmt"

// 1. Define a simple struct data type
type User struct {
	Name string
}

// 2. Attach a method to the struct using the receiver '(u User)'
func (u User) Greet() {
	// 'u' lets us access the data inside the struct instance
	fmt.Printf("Hello, my name is %s!\n", u.Name)
}

func main() {
	// 3. Create a new instance of our struct
	var admin User
	admin = User{Name: "Alice"}

	// 4. Call the method directly on the variable
	admin.Greet()
	// Output: Hello, my name is Alice!
}

/* output
hello my name is alice
*/
