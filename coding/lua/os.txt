

-- various example os functions

-- Date and Time
local now = os.time()
print("Timestamp:", now)

print("Current Date:", os.date("%Y-%m-%d %H:%M:%S", now))

local dateTable = os.date("*t", now)
print("Current Year:", dateTable.year)

local past = os.time({year=2026, month=1, day=1})
local secondsPassed = os.difftime(now, past)
print("Seconds since New Year 2026:", secondsPassed)

-- Output:textTimestamp: 1779459180
Current Date: 2026-05-22 14:13:00
Current Year: 2026
Seconds since New Year 2026: 12233580

-- System Control
local success, exitType, code = os.execute("echo 'Hello System'")
print("Command success:", success, "Type:", exitType, "Code:", code)

local home = os.getenv("HOME") or os.getenv("USERPROFILE")
print("User Home Directory:", home)

-- Output:
Hello System
Command success: true Type: exit Code: 0
User Home Directory: /home/user

-- File Management
-- Assumes 'old_name.txt' exists, and 'cleanup.txt' does not

local renamed, err = os.rename("old_name.txt", "new_name.txt")
print("Rename status:", renamed, err)

local deleted, delErr = os.remove("cleanup.txt")
print("Delete status:", deleted, delErr)

local tempPath = os.tmpname()
print("Temporary file path:", tempPath)

-- Output:
Rename status: true nil
Delete status: nil cleanup.txt: No such file or directory
Temporary file path: /tmp/lua_3X8aBc

os.setlocale()
-- Sets or queries the current localization configuration of the underlying operating system.
-- This alters how characters, numbers, and dates are formatted or sorted.

-- Syntax: os.setlocale(locale [, category])
-- Common Categories: "all", "collate", "ctype", "monetary", "numeric", "time".

-- Set locale to French for date and time formatting

local success = os.setlocale("fr_FR.UTF-8", "time")
print("Locale set success:", success)

-- Printing the date will now reflect the new locale configuration
if success then
    print("French Month Name:", os.date("%B"))
end

-- Reset back to system default locale
os.setlocale("", "all")

Output:

Locale set success:	fr_FR.UTF-8
French Month Name:	mai

-- Passing Arguments to os.exit()
-- While os.exit() was shown previously to terminate programs,
-- it supports an optional second parameter close (introduced in Lua 5.2)
-- which forces the Lua state to safely close its garbage collector and
-- run all pending object destructors before exiting.
-- os.exit([code [, close]])

print("This line will print.")

-- Exit with a 'failure' status code (1) and force close the Lua state safely
os.exit(1, true)

print("This line will never be reached.")

-- output:
This line will print.

-- Passing Arguments to os.time()When called without arguments,
-- os.time() returns the current timestamp.
-- However, passing a formatted dictionary table allows you to calculate
-- timestamps for specific historic or future dates.

-- Syntax: os.time([table])
-- Required Keys: year, month, day
-- Optional Keys: hour (defaults to 12), min (defaults to 0), sec (defaults to 0),
-- isdst (Daylight Saving Boolean)

-- Generate a timestamp for exact time: October 25, 2028, at 08:30:00 AM
local customTime = os.time({
    year = 2028,
    month = 10,
    day = 25,
    hour = 8,
    min = 30,
    sec = 0
})

print("Generated Epoch Timestamp:", customTime)
print("Verified via os.date:", os.date("%Y-%m-%d %H:%M:%S", customTime))

-- Output:
Generated Epoch Timestamp: 1856068200
Verified via os.date: 2028-10-25 08:30:00
