

... use _if to check and determine dependency

from pyinfra.operations import apt, server, files

... install nginx

install_nginx = apt.packages(
    name="Install nginx",
    packages=["nginx"],
    update=True,
    _sudo=yes,
)

... capture the operation result
... place config file only if nginx package is installed

nginx_config = files.template(
    name="upload nginx config",
    src="templates/nginx.conf.j2",
    dest="/etc/nginx/nginx.conf",
    _if=install_nginx.did_change,
)

... then restart only if the config file actually changed

server.service(
    name="Reload nginx",
    service="nginx",
    reloaded=True,
    _sudo=True,
    _if=nginx_config.did_change,
)
