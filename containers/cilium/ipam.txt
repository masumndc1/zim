-- delete any existing cluster first

❯ kind delete cluster --name kind
Deleting cluster "kind" ...

-- recreate again

> kind create cluster --config kind.yaml

-- install cilium again with custome settings

❯ cat  cilium-ipam-cluster-scope.yaml

  name: kind-kind

  mode: cluster-pool
operator:
  replicas: 1
routingMode: tunnel
tunnelProtocol: vxlan

-- when ipam is multi-pool

cat cilium-ipam-multi-pool.yaml
ipam:
  mode: multi-pool
  operator:
    autoCreateCiliumPodIPPools:
      default:
        ipv4:
          cidrs: ["10.10.0.0/16"]
          maskSize: 27

routingMode: native
endpointRoutes:
  enabled: true

autoDirectNodeRoutes: true
ipv4NativeRoutingCIDR: 10.0.0.0/8

-- lets create following two pools

❯ cat acme-pool.yaml
apiVersion: cilium.io/v2alpha1
kind: CiliumPodIPPool
metadata:
  name: acme-pool
spec:
  ipv4:
    cidrs:
    - 10.20.0.0/16
    maskSize: 27

❯ cat foobar-pool.yaml
apiVersion: cilium.io/v2alpha1
kind: CiliumPodIPPool
metadata:
  name: foobar-pool
spec:
  ipv4:
    cidrs:
    - 10.30.0.0/16
    maskSize: 27

-- annotate them

❯ kubectl annotate ns acme-corp ipam.cilium.io/ip-pool=acme-pool
namespace/acme-corp annotated
❯ kubectl annotate ns foobar-inc ipam.cilium.io/ip-pool=foobar-pool
namespace/foobar-inc annotated
