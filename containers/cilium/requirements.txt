


--
-- requirements for gateway api in cilium.
--
-- the cluster should have following settings
-- where kubeProxyMode should be none. we have to disable this.
-- moreover, as we are going to use cilium, thefore, we need to
-- disable DefaultCNI.

kind: Cluster
apiVersion: kind.x-k8s.io/v1alpha4
nodes:
  - role: control-plane
  - role: worker
  - role: worker
networking:
  kubeProxyMode: none
  disableDefaultCNI: true
  podSubnet: "10.244.0.0/16" // this is optional here.

-- If you are installing cilium with helm, then following options
-- are important.

kubeProxyReplacement: true
l2announcements:
  enabled: true
devices:
  - eth0
ipam:
  mode: kubernetes
gatewayAPI:
  enabled: true

--
-- then, look at the following yaml file to configure gateway api with cilium.
-- in basic setup only forllowing gateway resources needed to activate cilium.
-- they are Gateway, CiliumLoadBalancerIPPool, CiliumL2AnnouncementPolicy.
--

> cat yaml_files/gateway_api.yaml
---
# gateway
apiVersion: gateway.networking.k8s.io/v1
kind: Gateway
metadata:
  name: cilium-gateway
  namespace: kube-system
spec:
  gatewayClassName: cilium
  listeners:
    - protocol: HTTP
      port: 80
      name: web-gw
      allowedRoutes:
        namespaces:
          from: All
---
# load balancer pool
apiVersion: "cilium.io/v2"
kind: CiliumLoadBalancerIPPool
metadata:
  name: "lb-pool"
spec:
  blocks:
  - start: 172.18.18.1
    stop: 172.18.18.254
---
# l2 announcement policy
apiVersion: "cilium.io/v2alpha1"
kind: CiliumL2AnnouncementPolicy
metadata:
  name: policy1
spec:
  loadBalancerIPs: true
  interfaces:
    - eth0


-- then we need to define service kind for the depolyment.
-- in Service, you should not define any ClusterIP or LoadBalancer
-- type in Service, because this will be handled by gateway API.

cat yaml_files/nginx.yaml
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx
  namespace: default
spec:
  replicas: 3
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
        app.kubernetes.io/name: nginx
    spec:
      containers:
        - name: nginx
          image: nginx:alpine
---
apiVersion: v1
kind: Service
metadata:
  name: nginx-service
  namespace: default
spec:
  selector:
    app.kubernetes.io/name: nginx
  ports:
    - port: 8080
      targetPort: 80
      name: http
---
apiVersion: gateway.networking.k8s.io/v1
kind: HTTPRoute
metadata:
  name: nginx-route
  namespace: default
spec:
  parentRefs:
   - name: cilium-gateway
     namespace: kube-system
  hostnames:
   - "nginx.masum.com"
  rules:
   - backendRefs:
     - kind: Service
       name: nginx-service
       port: 8080
