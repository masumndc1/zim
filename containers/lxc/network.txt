

... networking in incus/lxc
...
... 2nd network for incus


incus network create incusbr1 \
  ipv4.address=10.20.0.1/24 \
  ipv4.nat=true \
  ipv6.address=none

... attach it to a containers.

incus network attach incusbr1 mycontainer eth1
