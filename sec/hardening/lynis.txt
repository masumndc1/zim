-- various scan options

sudo lynis audit system
sudo lynis audit system --quick
sudo lynis audit system --cronjob // this will reduce coloring output

-- run on docker containers running in the node
lynis audit dockerfile

-- remote system
lynis audit system remote <remote_hostname>

-- Limits the audit scope strictly to specific categories to save scan time.
sudo lynis audit system --tests-from-category "firewalls authentication"
