-- before you enforce any policy in kubernetes, we need to find out PODCIDR and serviceCIDR

export PODCIDR=`kubectl get nodes -o jsonpath='{.items[*].spec.podCIDR}'`
export SERVICECIDR=$(kubectl describe pod -n kube-system kube-apiserver-kind-control-plane | awk -F= '/--service-cluster-ip-range/ {print $2; }')

-- then run following to enforce policy in egress traffic
-- before that we need to delete any existing similar egress cluster there.

❯ kubectl delete tracingpolicy monitor-network-activity-outside-cluster-cidr-range
tracingpolicy.cilium.io "monitor-network-activity-outside-cluster-cidr-range" deleted

-- now run following
envsubst < network_egress_cluster_enforce.yaml | kubectl apply -n default -f -
