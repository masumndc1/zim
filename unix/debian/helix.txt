... how to setup language for python in helix
...
... install ruff and ty by using pip
... ruff is for: linter and formatter
... ty is for: lsp

myenv > pip3 install ruff ty

❯ pwd
~/.config/helix
❯ ls
 config.toml   languages.toml

❯ cat language.toml
[[language]]
name = "python"
language-servers = ["ty","ruff"]
auto-format = true

[language-server.ty]
command = "ty"
args = ["server"]

[language-server.ruff]
command = "ruff"
args = ["server"]

[language.formatter]
command = "ruff"
args = ["format", "-"]
