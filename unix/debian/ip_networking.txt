how to change the hostname in redhat5.

[root@ sysconfig]# pwd
/etc/sysconfig
[root@ sysconfig]# vi network
[root@ sysconfig]# cat -n network
     1	NETWORKING=yes
     2	NETWORKING_IPV6=no
     3	HOSTNAME=mail.example.com
[root@ sysconfig]#
