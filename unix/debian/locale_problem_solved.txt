masum@ipaclient:~$ cat /etc/default/locale 
#  File generated by update-locale
LANG=en_US.UTF-8
masum@ipaclient:~$ 
masum@ipaclient:~$ sudo update-locale LC_ALL=en_US.UTF-8 LANG=en_US.UTF-8
masum@ipaclient:~$ 
masum@ipaclient:~$ cat /etc/default/locale 
#  File generated by update-locale
LANG=en_US.UTF-8
LC_ALL=en_US.UTF-8
masum@ipaclient:~$
