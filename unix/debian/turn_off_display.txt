
--
-- https://askubuntu.com/questions/62858/turn-off-monitor-using-command-line
-- in debian/centos turn of monitor/display by following.
--

-- create a script:

/usr/local/bin/screen-off.sh

#!/bin/bash
setterm --blank 1 --powerdown 2


-- create a system service like below

/etc/systemd/system/screen-off.service

[Unit]
Description=Blank screen after 1 min and turn it off after 2 min. Any keypress will turn it back on.
After=ssh.service

[Service]
Type=oneshot
Environment=TERM=linux
StandardOutput=tty
TTYPath=/dev/console
ExecStart=/usr/local/bin/screen-off.sh

[Install]
WantedBy=local.target

-- give it executive right
sudo chmod +x /usr/local/bin/screen-off.sh

-- administer it by following
-- enable it for boot:

sudo systemctl enable screen-off.service

-- start the service right away:
sudo systemctl start screen-off.service

-- disable it:
sudo systemctl disable screen-off.service
