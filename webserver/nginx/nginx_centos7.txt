... installing nginx through puppet

[root@puppet manifest]# cat nginx.pp
package { 'nginx':
  ensure => 'present',
}
service { 'nginx':
  ensure => 'running',
  enable => 'true',
}

exec { 'query':
	path => '/usr/bin',
	command => 'rpm -qa | grep nginx',
	logoutput => 'true',
}
[root@puppet manifest]#
